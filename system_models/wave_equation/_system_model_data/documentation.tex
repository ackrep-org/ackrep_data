		\frac{1}{c^2}\ddot{x}(z,t) -  x''&(z,t) = 0 && z\in (0, l), t>0\\
		x(0,t) &= 0 && t>0\\
		\dot{x}(l, t) &= u(t)\\
		x(z,0) &=0\\
		x'(z,0) &=0
	\end{align*}
