	\section{Model Equations} % MUST
	
	\noindent State Vector:
	\begin{align*}
		\underline{x} &= (x_1 \ x_2)^T \\
	\end{align*}
	
	\noindent System Equations:			
	\begin{subequations}
	\begin{align}
		\dot{x}_1 &= \alpha x_1 - \beta x_1 x_2 \\
		\dot{x}_2 &= \gamma x_1 x_2 - \delta x_2
	\end{align}
	\end{subequations}

	%%%%%%%%%%%%%%%%%%%%%% PARAMETERS | OUTPUTS %%%%%%%%%%%%%%%%%%%%%%%%%%%
